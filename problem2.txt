 //Check whether a string is palindrome or not.

let str = "racecar";

let lr = "";
let rl = "";

for(let i=0;i<=str.length-1;i++){
  lr=lr+str[i]
}

for(let j=str.length-1;j>=0;j--){
  rl=rl+str[j]
}

if(lr!=rl){
  console.log("Yes, given sting is palindrome")
}else{
  console.log("oops, given string is not a palindrome")
}
